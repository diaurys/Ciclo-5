Escribir una función entera que devuelva el número de nodos de una lista enlazada

typedef struct Nodo {
  int dato;
  struct Nodo* siguiente;
} Nodo;

int contarNodos(Nodo* cabeza) {
  int contador = 0;
  Nodo* actual = cabeza;

  while (actual != NULL) {
    contador++;
    actual = actual->siguiente;
  }

  return contador;
}
